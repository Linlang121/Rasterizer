Texture2D tex : register(t0);
Texture2D normalMap : register(t1);
SamplerState samplerLinear : register(s0);

struct PS_INPUT
{
    float4 Pos : SV_POSITION;
    float3 Normal : NORMAL;
    float3 Tangent : TANGENT;
    float2 TexCoords : TEXCOORD;
};

float4 PS(PS_INPUT input) : SV_Target0
{
    float4 colour = tex.Sample(samplerLinear, input.TexCoords);
    clip(colour.a - 0.01);
    
    float3 N = normalize(input.Normal);
    float3 worldNormal = N;
    
    float3 normalMapSample = normalMap.Sample(samplerLinear, input.TexCoords).rgb;
    if (length(normalMapSample) > 0.1)
    {
        float3 tangentNormal = normalize(normalMapSample * 2.0 - 1.0);
        float3 T = normalize(input.Tangent);
        float3 B = cross(N, T);
        float3x3 TBN = float3x3(T, B, N);
        worldNormal = mul(tangentNormal, TBN);
        worldNormal = normalize(worldNormal);
    }
    
    float3 lightDir = normalize(float3(1.0, 1.0, 1.0));
    float NdotL = max(dot(worldNormal, lightDir), 0.0);
    
    colour.rgb *= (0.3 + 0.7 * NdotL);
    colour.rgb = pow(colour.rgb, 1.0 / 2.2);
    
    return colour;
}
